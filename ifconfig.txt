ifconfig is used to show the ip,mac,dev and show stuff about network
ifconfig #print normal information
ifconfig eth0 or wlan0 #print specific interface address
ifconfig wlan0 | grep -w inet #output: inet 192.168.43.163  netmask 255.255.255.0  broadcast 192.168.43.255}
ifconfig wlan0 | grep -w inet |awk {'print $2'} #output:192.168.43.163
ifconfig wlan0 down #To down the wireless drive
ifconfig wlan0 up #To up the wireless drive
